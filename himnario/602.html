Ya resuena alegre la canción,

Para siempre santidad;

Nos anuncia plena salvación,

Para siempre santidad.

CORO

// Para siempre santidad //

Alabemos todos al Rey de gloria,

Para siempre santidad.

2

Del pecado innato nos limpió,

Para siempre santidad;

De su amor perfecto, nos llenó,

Para siempre santidad.

CORO

// Para siempre santidad //

Alabemos todos al Rey de gloria,

Para siempre santidad.

3

Alzaremos fieles el pendón,

Para siempre santidad;

Lucharemos sin vacilación,

Para siempre santidad.

CORO

// Para siempre santidad //

Alabemos todos al Rey de gloria,

Para siempre santidad.

4

Alabanzas demos al Señor,

Para siempre santidad;

Entonemos todos al Señor,

Para siempre santidad.

CORO

// Para siempre santidad //

Alabemos todos al Rey de gloria,

Para siempre santidad.
